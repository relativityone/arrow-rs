21const BASIC_TYPE_BITS: u8 = 2;
22const MAX_SHORT_STRING_SIZE: usize = 0x3F;
23const UNIX_EPOCH_DATE: chrono::NaiveDate = chrono::NaiveDate::from_ymd_opt(1970, 1, 1).unwrap();
24
25fn primitive_header(primitive_type: VariantPrimitiveType) -> u8 {
26    (primitive_type as u8) << 2 | VariantBasicType::Primitive as u8
27}
28
29fn short_string_header(len: usize) -> u8 {
30    (len as u8) << 2 | VariantBasicType::ShortString as u8
31}
32
33fn array_header(large: bool, offset_size: u8) -> u8 {
34    let large_bit = if large { 1 } else { 0 };
35    (large_bit << (BASIC_TYPE_BITS + 2))
36        | ((offset_size - 1) << BASIC_TYPE_BITS)
37        | VariantBasicType::Array as u8
38}
39
40fn object_header(large: bool, id_size: u8, offset_size: u8) -> u8 {
41    let large_bit = if large { 1 } else { 0 };
42    (large_bit << (BASIC_TYPE_BITS + 4))
43        | ((id_size - 1) << (BASIC_TYPE_BITS + 2))
44        | ((offset_size - 1) << BASIC_TYPE_BITS)
45        | VariantBasicType::Object as u8
46}
47
48fn int_size(v: usize) -> u8 {
49    match v {
50        0..=0xFF => 1,
51        0x100..=0xFFFF => 2,
52        0x10000..=0xFFFFFF => 3,
53        _ => 4,
54    }
55}
56
57/// Write little-endian integer to buffer
58fn write_offset(buf: &mut [u8], value: usize, nbytes: u8) {
59    for i in 0..nbytes {
60        buf[i as usize] = (value >> (i * 8)) as u8;
61    }
62}
63
64/// Helper to make room for header by moving data
65fn make_room_for_header(buffer: &mut Vec<u8>, start_pos: usize, header_size: usize) {
66    let current_len = buffer.len();
67    buffer.resize(current_len + header_size, 0);
68
69    let src_start = start_pos;
70    let src_end = current_len;
71    let dst_start = start_pos + header_size;
72
73    buffer.copy_within(src_start..src_end, dst_start);
74}

76/// Builder for [`Variant`] values
77///
78/// # Example: create a Primitive Int8
79/// ```
80/// # use parquet_variant::{Variant, VariantBuilder};
81/// let mut builder = VariantBuilder::new();
82/// builder.append_value(Variant::Int8(42));
83/// // Finish the builder to get the metadata and value
84/// let (metadata, value) = builder.finish();
85/// // use the Variant API to verify the result
86/// let variant = Variant::try_new(&metadata, &value).unwrap();
87/// assert_eq!(variant, Variant::Int8(42));
88/// ```
89///
90/// # Example: Create a [`Variant::Object`]
91///
92/// This example shows how to create an object with two fields:
93/// ```json
94/// {
95///  "first_name": "Jiaying",
96///  "last_name": "Li"
97/// }
98/// ```
99///
100/// ```
101/// # use parquet_variant::{Variant, VariantBuilder};
102/// let mut builder = VariantBuilder::new();
103/// // Create an object builder that will write fields to the object
104/// let mut object_builder = builder.new_object();
105/// object_builder.append_value("first_name", "Jiaying");
106/// object_builder.append_value("last_name", "Li");
107/// object_builder.finish();
108/// // Finish the builder to get the metadata and value
109/// let (metadata, value) = builder.finish();
110/// // use the Variant API to verify the result
111/// let variant = Variant::try_new(&metadata, &value).unwrap();
112/// let Variant::Object(variant_object) = variant else {
113///   panic!("unexpected variant type")
114/// };
115/// assert_eq!(
116///   variant_object.field("first_name").unwrap(),
117///   Some(Variant::ShortString("Jiaying"))
118/// );
119/// assert_eq!(
120///   variant_object.field("last_name").unwrap(),
121///   Some(Variant::ShortString("Li"))
122/// );
123/// ```
124///
125/// # Example: Create a [`Variant::List`] (an Array)
126///
127/// This example shows how to create an array of integers: `[1, 2, 3]`.
128/// ```
129///  # use parquet_variant::{Variant, VariantBuilder};
130///  let mut builder = VariantBuilder::new();
131///  // Create a builder that will write elements to the list
132///  let mut list_builder = builder.new_list();
133///  list_builder.append_value(1i8);
134///  list_builder.append_value(2i8);
135///  list_builder.append_value(3i8);
136///  list_builder.finish();
137/// // Finish the builder to get the metadata and value
138/// let (metadata, value) = builder.finish();
139/// // use the Variant API to verify the result
140/// let variant = Variant::try_new(&metadata, &value).unwrap();
141/// let Variant::List(variant_list) = variant else {
142///   panic!("unexpected variant type")
143/// };
144/// // Verify the list contents
145/// assert_eq!(variant_list.get(0).unwrap(), Variant::Int8(1));
146/// assert_eq!(variant_list.get(1).unwrap(), Variant::Int8(2));
147/// assert_eq!(variant_list.get(2).unwrap(), Variant::Int8(3));
148/// ```
149///
150/// # Example: [`Variant::List`] of  [`Variant::Object`]s
151///
152/// THis example shows how to create an list  of objects:
153/// ```json
154/// [
155///  {
156///   "first_name": "Jiaying",
157///  "last_name": "Li"
158/// },
159///   {
160///    "first_name": "Malthe",
161///    "last_name": "Karbo"
162/// }
163/// ]
164/// ```
165///
166/// TODO
167///
168pub struct VariantBuilder {
169    buffer: Vec<u8>,
170    dict: HashMap<String, u32>,
171    dict_keys: Vec<String>,
172}

174impl VariantBuilder {
175    pub fn new() -> Self {
176        Self {
177            buffer: Vec::new(),
178            dict: HashMap::new(),
179            dict_keys: Vec::new(),
180        }
181    }
182
183    fn append_null(&mut self) {
184        self.buffer
185            .push(primitive_header(VariantPrimitiveType::Null));
186    }
187
188    fn append_bool(&mut self, value: bool) {
189        let primitive_type = if value {
190            VariantPrimitiveType::BooleanTrue
191        } else {
192            VariantPrimitiveType::BooleanFalse
193        };
194        self.buffer.push(primitive_header(primitive_type));
195    }
196
197    fn append_int8(&mut self, value: i8) {
198        self.buffer
199            .push(primitive_header(VariantPrimitiveType::Int8));
200        self.buffer.push(value as u8);
201    }
202
203    fn append_int16(&mut self, value: i16) {
204        self.buffer
205            .push(primitive_header(VariantPrimitiveType::Int16));
206        self.buffer.extend_from_slice(&value.to_le_bytes());
207    }
208
209    fn append_int32(&mut self, value: i32) {
210        self.buffer
211            .push(primitive_header(VariantPrimitiveType::Int32));
212        self.buffer.extend_from_slice(&value.to_le_bytes());
213    }
214
215    fn append_int64(&mut self, value: i64) {
216        self.buffer
217            .push(primitive_header(VariantPrimitiveType::Int64));
218        self.buffer.extend_from_slice(&value.to_le_bytes());
219    }
220
221    fn append_float(&mut self, value: f32) {
222        self.buffer
223            .push(primitive_header(VariantPrimitiveType::Float));
224        self.buffer.extend_from_slice(&value.to_le_bytes());
225    }
226
227    fn append_double(&mut self, value: f64) {
228        self.buffer
229            .push(primitive_header(VariantPrimitiveType::Double));
230        self.buffer.extend_from_slice(&value.to_le_bytes());
231    }
232
233    fn append_date(&mut self, value: chrono::NaiveDate) {
234        self.buffer
235            .push(primitive_header(VariantPrimitiveType::Date));
236        let days_since_epoch = value.signed_duration_since(UNIX_EPOCH_DATE).num_days() as i32;
237        self.buffer
238            .extend_from_slice(&days_since_epoch.to_le_bytes());
239    }
240
241    fn append_timestamp_micros(&mut self, value: chrono::DateTime<chrono::Utc>) {
242        self.buffer
243            .push(primitive_header(VariantPrimitiveType::TimestampMicros));
244        let micros = value.timestamp_micros();
245        self.buffer.extend_from_slice(&micros.to_le_bytes());
246    }
247
248    fn append_timestamp_ntz_micros(&mut self, value: chrono::NaiveDateTime) {
249        self.buffer
250            .push(primitive_header(VariantPrimitiveType::TimestampNtzMicros));
251        let micros = value.and_utc().timestamp_micros();
252        self.buffer.extend_from_slice(&micros.to_le_bytes());
253    }
254
255    fn append_decimal4(&mut self, integer: i32, scale: u8) {
256        self.buffer
257            .push(primitive_header(VariantPrimitiveType::Decimal4));
258        self.buffer.push(scale);
259        self.buffer.extend_from_slice(&integer.to_le_bytes());
260    }
261
262    fn append_decimal8(&mut self, integer: i64, scale: u8) {
263        self.buffer
264            .push(primitive_header(VariantPrimitiveType::Decimal8));
265        self.buffer.push(scale);
266        self.buffer.extend_from_slice(&integer.to_le_bytes());
267    }
268
269    fn append_decimal16(&mut self, integer: i128, scale: u8) {
270        self.buffer
271            .push(primitive_header(VariantPrimitiveType::Decimal16));
272        self.buffer.push(scale);
273        self.buffer.extend_from_slice(&integer.to_le_bytes());
274    }
275
276    fn append_binary(&mut self, value: &[u8]) {
277        self.buffer
278            .push(primitive_header(VariantPrimitiveType::Binary));
279        self.buffer
280            .extend_from_slice(&(value.len() as u32).to_le_bytes());
281        self.buffer.extend_from_slice(value);
282    }
283
284    fn append_string(&mut self, value: &str) {
285        if value.len() <= MAX_SHORT_STRING_SIZE {
286            self.buffer.push(short_string_header(value.len()));
287            self.buffer.extend_from_slice(value.as_bytes());
288        } else {
289            self.buffer
290                .push(primitive_header(VariantPrimitiveType::String));
291            self.buffer
292                .extend_from_slice(&(value.len() as u32).to_le_bytes());
293            self.buffer.extend_from_slice(value.as_bytes());
294        }
295    }
296
297    /// Add key to dictionary, return its ID
298    fn add_key(&mut self, key: &str) -> u32 {
299        use std::collections::hash_map::Entry;
300        match self.dict.entry(key.to_string()) {
301            Entry::Occupied(entry) => *entry.get(),
302            Entry::Vacant(entry) => {
303                let id = self.dict_keys.len() as u32;
304                entry.insert(id);
305                self.dict_keys.push(key.to_string());
306                id
307            }
308        }
309    }
310
311    fn offset(&self) -> usize {
312        self.buffer.len()
313    }

318    pub fn new_list(&mut self) -> ListBuilder {
319        ListBuilder::new(self)
320    }

323    ///
324    /// See the examples on [`VariantBuilder`] for usage.
325    pub fn new_object(&mut self) -> ObjectBuilder {
326        ObjectBuilder::new(self)
327    }

329    pub fn finish(self) -> (Vec<u8>, Vec<u8>) {
330        let nkeys = self.dict_keys.len();
331
332        // Calculate metadata size
333        let total_dict_size: usize = self.dict_keys.iter().map(|k| k.len()).sum();
334
335        // Determine appropriate offset size based on the larger of dict size or total string size
336        let max_offset = std::cmp::max(total_dict_size, nkeys);
337        let offset_size = int_size(max_offset);
338
339        let offset_start = 1 + offset_size as usize;
340        let string_start = offset_start + (nkeys + 1) * offset_size as usize;
341        let metadata_size = string_start + total_dict_size;
342
343        // Pre-allocate exact size to avoid reallocations
344        let mut metadata = vec![0u8; metadata_size];
345
346        // Write header: version=1, not sorted, with calculated offset_size
347        metadata[0] = 0x01 | ((offset_size - 1) << 6);
348
349        // Write dictionary size
350        write_offset(&mut metadata[1..], nkeys, offset_size);
351
352        // Write offsets and string data
353        let mut cur_offset = 0;
354        for (i, key) in self.dict_keys.iter().enumerate() {
355            write_offset(
356                &mut metadata[offset_start + i * offset_size as usize..],
357                cur_offset,
358                offset_size,
359            );
360            let start = string_start + cur_offset;
361            metadata[start..start + key.len()].copy_from_slice(key.as_bytes());
362            cur_offset += key.len();
363        }
364        // Write final offset
365        write_offset(
366            &mut metadata[offset_start + nkeys * offset_size as usize..],
367            cur_offset,
368            offset_size,
369        );
370
371        (metadata, self.buffer)
372    }
373
374    pub fn append_value<'m, 'd, T: Into<Variant<'m, 'd>>>(&mut self, value: T) {
375        let variant = value.into();
376        match variant {
377            Variant::Null => self.append_null(),
378            Variant::BooleanTrue => self.append_bool(true),
379            Variant::BooleanFalse => self.append_bool(false),
380            Variant::Int8(v) => self.append_int8(v),
381            Variant::Int16(v) => self.append_int16(v),
382            Variant::Int32(v) => self.append_int32(v),
383            Variant::Int64(v) => self.append_int64(v),
384            Variant::Date(v) => self.append_date(v),
385            Variant::TimestampMicros(v) => self.append_timestamp_micros(v),
386            Variant::TimestampNtzMicros(v) => self.append_timestamp_ntz_micros(v),
387            Variant::Decimal4 { integer, scale } => self.append_decimal4(integer, scale),
388            Variant::Decimal8 { integer, scale } => self.append_decimal8(integer, scale),
389            Variant::Decimal16 { integer, scale } => self.append_decimal16(integer, scale),
390            Variant::Float(v) => self.append_float(v),
391            Variant::Double(v) => self.append_double(v),
392            Variant::Binary(v) => self.append_binary(v),
393            Variant::String(s) | Variant::ShortString(s) => self.append_string(s),
394            Variant::Object(_) | Variant::List(_) => {
395                unreachable!("Object and List variants cannot be created through Into<Variant>")
396            }
397        }
398    }
399}

407/// A builder for creating [`Variant::List`] values.
408///
409/// See the examples on [`VariantBuilder`] for usage.
410pub struct ListBuilder<'a> {
411    parent: &'a mut VariantBuilder,
412    start_pos: usize,
413    offsets: Vec<usize>,
414}
415
416impl<'a> ListBuilder<'a> {
417    fn new(parent: &'a mut VariantBuilder) -> Self {
418        let start_pos = parent.offset();
419        Self {
420            parent,
421            start_pos,
422            offsets: vec![0],
423        }
424    }
425
426    pub fn append_value<'m, 'd, T: Into<Variant<'m, 'd>>>(&mut self, value: T) {
427        self.parent.append_value(value);
428        let element_end = self.parent.offset() - self.start_pos;
429        self.offsets.push(element_end);
430    }
431
432    pub fn finish(self) {
433        let data_size = self.parent.offset() - self.start_pos;
434        let num_elements = self.offsets.len() - 1;
435        let is_large = num_elements > u8::MAX as usize;
436        let size_bytes = if is_large { 4 } else { 1 };
437        let offset_size = int_size(data_size);
438        let header_size = 1 + size_bytes + (num_elements + 1) * offset_size as usize;
439
440        make_room_for_header(&mut self.parent.buffer, self.start_pos, header_size);
441
442        // Write header
443        let mut pos = self.start_pos;
444        self.parent.buffer[pos] = array_header(is_large, offset_size);
445        pos += 1;
446
447        if is_large {
448            self.parent.buffer[pos..pos + 4].copy_from_slice(&(num_elements as u32).to_le_bytes());
449            pos += 4;
450        } else {
451            self.parent.buffer[pos] = num_elements as u8;
452            pos += 1;
453        }
454
455        // Write offsets
456        for offset in &self.offsets {
457            write_offset(
458                &mut self.parent.buffer[pos..pos + offset_size as usize],
459                *offset,
460                offset_size,
461            );
462            pos += offset_size as usize;
463        }
464    }
465}

467/// A builder for creating [`Variant::Object`] values.
468///
469/// See the examples on [`VariantBuilder`] for usage.
470pub struct ObjectBuilder<'a> {
471    parent: &'a mut VariantBuilder,
472    start_pos: usize,
473    fields: Vec<(u32, usize)>, // (field_id, offset)
474}
475
476impl<'a> ObjectBuilder<'a> {
477    fn new(parent: &'a mut VariantBuilder) -> Self {
478        let start_pos = parent.offset();
479        Self {
480            parent,
481            start_pos,
482            fields: Vec::new(),
483        }
484    }
485
486    /// Add a field with key and value to the object
487    pub fn append_value<'m, 'd, T: Into<Variant<'m, 'd>>>(&mut self, key: &str, value: T) {
488        let id = self.parent.add_key(key);
489        let field_start = self.parent.offset() - self.start_pos;
490        self.parent.append_value(value);
491        self.fields.push((id, field_start));
492    }
493
494    /// Finalize object with sorted fields
495    pub fn finish(mut self) {
496        // Sort fields by key name
497        self.fields.sort_by(|a, b| {
498            let key_a = &self.parent.dict_keys[a.0 as usize];
499            let key_b = &self.parent.dict_keys[b.0 as usize];
500            key_a.cmp(key_b)
501        });
502
503        let data_size = self.parent.offset() - self.start_pos;
504        let num_fields = self.fields.len();
505        let is_large = num_fields > u8::MAX as usize;
506        let size_bytes = if is_large { 4 } else { 1 };
507
508        let max_id = self.fields.iter().map(|&(id, _)| id).max().unwrap_or(0);
509        let id_size = int_size(max_id as usize);
510        let offset_size = int_size(data_size);
511
512        let header_size = 1
513            + size_bytes
514            + num_fields * id_size as usize
515            + (num_fields + 1) * offset_size as usize;
516
517        make_room_for_header(&mut self.parent.buffer, self.start_pos, header_size);
518
519        // Write header
520        let mut pos = self.start_pos;
521        self.parent.buffer[pos] = object_header(is_large, id_size, offset_size);
522        pos += 1;
523
524        if is_large {
525            self.parent.buffer[pos..pos + 4].copy_from_slice(&(num_fields as u32).to_le_bytes());
526            pos += 4;
527        } else {
528            self.parent.buffer[pos] = num_fields as u8;
529            pos += 1;
530        }
531
532        // Write field IDs (sorted order)
533        for &(id, _) in &self.fields {
534            write_offset(
535                &mut self.parent.buffer[pos..pos + id_size as usize],
536                id as usize,
537                id_size,
538            );
539            pos += id_size as usize;
540        }
541
542        // Write field offsets
543        for &(_, offset) in &self.fields {
544            write_offset(
545                &mut self.parent.buffer[pos..pos + offset_size as usize],
546                offset,
547                offset_size,
548            );
549            pos += offset_size as usize;
550        }
551        write_offset(
552            &mut self.parent.buffer[pos..pos + offset_size as usize],
553            data_size,
554            offset_size,
555        );
556    }
557}

716
717        {
718            let mut obj = builder.new_object();
719            obj.append_value("zebra", "stripes"); // ID = 0
720            obj.append_value("apple", "red"); // ID = 1
721            obj.append_value("banana", "yellow"); // ID = 2
722            obj.finish();
723        }
724
725        let (_, value) = builder.finish();
726
727        let header = value[0];

729
730        let field_count = value[1] as usize;
731        assert_eq!(field_count, 3);
732
733        // Get field IDs from the object header
734        let field_ids: Vec<u8> = value[2..5].to_vec();
735
736        // apple(1), banana(2), zebra(0)
737        assert_eq!(field_ids, vec![1, 2, 0]);
