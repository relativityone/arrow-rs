28#[derive(Clone, Debug, Copy, PartialEq)]
29enum OffsetSizeBytes {
30    One = 1,
31    Two = 2,
32    Three = 3,
33    Four = 4,
34}
35
36impl OffsetSizeBytes {
37    /// Build from the `offset_size_minus_one` bits (see spec).
38    fn try_new(offset_size_minus_one: u8) -> Result<Self, ArrowError> {
39        use OffsetSizeBytes::*;
40        let result = match offset_size_minus_one {
41            0 => One,
42            1 => Two,
43            2 => Three,
44            3 => Four,
45            _ => {
46                return Err(ArrowError::InvalidArgumentError(
47                    "offset_size_minus_one must be 0–3".to_string(),
48                ))
49            }
50        };
51        Ok(result)
52    }
53
54    /// Return one unsigned little-endian value from `bytes`.
55    ///
56    /// * `bytes` – the Variant-metadata buffer.
57    /// * `byte_offset` – number of bytes to skip **before** reading the first
58    ///   value (usually `1` to move past the header byte).
59    /// * `offset_index` – 0-based index **after** the skip
60    ///   (`0` is the first value, `1` the next, …).
61    ///
62    /// Each value is `self as usize` bytes wide (1, 2, 3 or 4).
63    /// Three-byte values are zero-extended to 32 bits before the final
64    /// fallible cast to `usize`.
65    fn unpack_usize(
66        &self,
67        bytes: &[u8],
68        byte_offset: usize,  // how many bytes to skip
69        offset_index: usize, // which offset in an array of offsets
70    ) -> Result<usize, ArrowError> {
71        use OffsetSizeBytes::*;
72        let offset = byte_offset + (*self as usize) * offset_index;
73        let result = match self {
74            One => u8::from_le_bytes(array_from_slice(bytes, offset)?).into(),
75            Two => u16::from_le_bytes(array_from_slice(bytes, offset)?).into(),
76            Three => {
77                // Let's grab the three byte le-chunk first
78                let b3_chunks: [u8; 3] = array_from_slice(bytes, offset)?;
79                // Let's pad it and construct a padded u32 from it.
80                let mut buf = [0u8; 4];
81                buf[..3].copy_from_slice(&b3_chunks);
82                u32::from_le_bytes(buf)
83                    .try_into()
84                    .map_err(|e: TryFromIntError| ArrowError::InvalidArgumentError(e.to_string()))?
85            }
86            Four => u32::from_le_bytes(array_from_slice(bytes, offset)?)
87                .try_into()
88                .map_err(|e: TryFromIntError| ArrowError::InvalidArgumentError(e.to_string()))?,
89        };
90        Ok(result)
91    }
92}

94#[derive(Clone, Debug, Copy, PartialEq)]
95pub(crate) struct VariantMetadataHeader {
96    version: u8,
97    is_sorted: bool,
98    /// Note: This is `offset_size_minus_one` + 1
99    offset_size: OffsetSizeBytes,
100}
101
102// According to the spec this is currently always = 1, and so we store this const for validation
103// purposes and to make that visible.
104const CORRECT_VERSION_VALUE: u8 = 1;
105
106impl VariantMetadataHeader {
107    /// Tries to construct the variant metadata header, which has the form
108    ///              7     6  5   4  3             0
109    ///             +-------+---+---+---------------+
110    /// header      |       |   |   |    version    |
111    ///             +-------+---+---+---------------+
112    ///                 ^         ^
113    ///                 |         +-- sorted_strings
114    ///                 +-- offset_size_minus_one
115    /// The version is a 4-bit value that must always contain the value 1.
116    /// - sorted_strings is a 1-bit value indicating whether dictionary strings are sorted and unique.
117    /// - offset_size_minus_one is a 2-bit value providing the number of bytes per dictionary size and offset field.
118    /// - The actual number of bytes, offset_size, is offset_size_minus_one + 1
119    pub(crate) fn try_new(bytes: &[u8]) -> Result<Self, ArrowError> {
120        let header = first_byte_from_slice(bytes)?;
121
122        let version = header & 0x0F; // First four bits
123        if version != CORRECT_VERSION_VALUE {
124            let err_msg = format!(
125                "The version bytes in the header is not {CORRECT_VERSION_VALUE}, got {:b}",
126                version
127            );
128            return Err(ArrowError::InvalidArgumentError(err_msg));
129        }
130        let is_sorted = (header & 0x10) != 0; // Fifth bit
131        let offset_size_minus_one = header >> 6; // Last two bits
132        Ok(Self {
133            version,
134            is_sorted,
135            offset_size: OffsetSizeBytes::try_new(offset_size_minus_one)?,
136        })
137    }
138}

140#[derive(Clone, Copy, Debug, PartialEq)]
141/// Encodes the Variant Metadata, see the Variant spec file for more information
142pub struct VariantMetadata<'m> {
143    bytes: &'m [u8],
144    header: VariantMetadataHeader,
145    dict_size: usize,
146    dictionary_key_start_byte: usize,
147}
148
149impl<'m> VariantMetadata<'m> {
150    /// View the raw bytes (needed by very low-level decoders)
151    #[inline]
152    pub const fn as_bytes(&self) -> &'m [u8] {
153        self.bytes
154    }
155
156    pub fn try_new(bytes: &'m [u8]) -> Result<Self, ArrowError> {
157        let header = VariantMetadataHeader::try_new(bytes)?;
158        // Offset 1, index 0 because first element after header is dictionary size
159        let dict_size = header.offset_size.unpack_usize(bytes, 1, 0)?;
160
161        // Check that we have the correct metadata length according to dictionary_size, or return
162        // error early.
163        // Minimum number of bytes the metadata buffer must contain:
164        // 1 byte header
165        // + offset_size-byte `dictionary_size` field
166        // + (dict_size + 1) offset entries, each `offset_size` bytes. (Table size, essentially)
167        // 1 + offset_size + (dict_size + 1) * offset_size
168        // = (dict_size + 2) * offset_size + 1
169        let offset_size = header.offset_size as usize; // Cheap to copy
170
171        let dictionary_key_start_byte = dict_size
172            .checked_add(2)
173            .and_then(|n| n.checked_mul(offset_size))
174            .and_then(|n| n.checked_add(1))
175            .ok_or_else(|| ArrowError::InvalidArgumentError("metadata length overflow".into()))?;
176
177        if bytes.len() < dictionary_key_start_byte {
178            return Err(ArrowError::InvalidArgumentError(
179                "Metadata shorter than dictionary_size implies".to_string(),
180            ));
181        }
182
183        // Check that all offsets are monotonically increasing
184        let mut offsets = (0..=dict_size).map(|i| header.offset_size.unpack_usize(bytes, 1, i + 1));
185        let Some(Ok(mut end @ 0)) = offsets.next() else {
186            return Err(ArrowError::InvalidArgumentError(
187                "First offset is non-zero".to_string(),
188            ));
189        };
190
191        for offset in offsets {
192            let offset = offset?;
193            if end >= offset {
194                return Err(ArrowError::InvalidArgumentError(
195                    "Offsets are not monotonically increasing".to_string(),
196                ));
197            }
198            end = offset;
199        }
200
201        // Verify the buffer covers the whole dictionary-string section
202        if end > bytes.len() - dictionary_key_start_byte {
203            // `prev` holds the last offset seen still
204            return Err(ArrowError::InvalidArgumentError(
205                "Last offset does not equal dictionary length".to_string(),
206            ));
207        }
208
209        Ok(Self {
210            bytes,
211            header,
212            dict_size,
213            dictionary_key_start_byte,
214        })
215    }
216
217    /// Whether the dictionary keys are sorted and unique
218    pub fn is_sorted(&self) -> bool {
219        self.header.is_sorted
220    }
221
222    /// Get the dictionary size
223    pub fn dictionary_size(&self) -> usize {
224        self.dict_size
225    }
226    pub fn version(&self) -> u8 {
227        self.header.version
228    }
229
230    /// Helper method to get the offset start and end range for a key by index.
231    fn get_offsets_for_key_by(&self, index: usize) -> Result<Range<usize>, ArrowError> {
232        if index >= self.dict_size {
233            return Err(ArrowError::InvalidArgumentError(format!(
234                "Index {} out of bounds for dictionary of length {}",
235                index, self.dict_size
236            )));
237        }
238
239        // Skipping the header byte (setting byte_offset = 1) and the dictionary_size (setting offset_index +1)
240        let unpack = |i| self.header.offset_size.unpack_usize(self.bytes, 1, i + 1);
241        Ok(unpack(index)?..unpack(index + 1)?)
242    }
243
244    /// Get a single offset by index
245    pub fn get_offset_by(&self, index: usize) -> Result<usize, ArrowError> {
246        if index >= self.dict_size {
247            return Err(ArrowError::InvalidArgumentError(format!(
248                "Index {} out of bounds for dictionary of length {}",
249                index, self.dict_size
250            )));
251        }
252
253        // Skipping the header byte (setting byte_offset = 1) and the dictionary_size (setting offset_index +1)
254        let unpack = |i| self.header.offset_size.unpack_usize(self.bytes, 1, i + 1);
255        unpack(index)
256    }
257
258    /// Get the key-name by index
259    pub fn get_field_by(&self, index: usize) -> Result<&'m str, ArrowError> {
260        let offset_range = self.get_offsets_for_key_by(index)?;
261        self.get_field_by_offset(offset_range)
262    }
263
264    /// Gets the field using an offset (Range) - helper method to keep consistent API.
265    pub(crate) fn get_field_by_offset(&self, offset: Range<usize>) -> Result<&'m str, ArrowError> {
266        let dictionary_keys_bytes =
267            slice_from_slice(self.bytes, self.dictionary_key_start_byte..self.bytes.len())?;
268        let result = string_from_slice(dictionary_keys_bytes, offset)?;
269
270        Ok(result)
271    }
272
273    #[allow(unused)]
274    pub(crate) fn header(&self) -> VariantMetadataHeader {
275        self.header
276    }
277
278    /// Get the offsets as an iterator
279    pub fn offsets(&self) -> impl Iterator<Item = Result<Range<usize>, ArrowError>> + 'm {
280        let offset_size = self.header.offset_size; // `Copy`
281        let bytes = self.bytes;
282
283        (0..self.dict_size).map(move |i| {
284            // This wont be out of bounds as long as dict_size and offsets have been validated
285            // during construction via `try_new`, as it calls unpack_usize for the
286            // indices `1..dict_size+1` already.
287            let start = offset_size.unpack_usize(bytes, 1, i + 1);
288            let end = offset_size.unpack_usize(bytes, 1, i + 2);
289
290            match (start, end) {
291                (Ok(s), Ok(e)) => Ok(s..e),
292                (Err(e), _) | (_, Err(e)) => Err(e),
293            }
294        })
295    }
296
297    /// Get all key-names as an Iterator of strings
298    pub fn fields(
299        &'m self,
300    ) -> Result<impl Iterator<Item = Result<&'m str, ArrowError>>, ArrowError> {
301        let iterator = self
302            .offsets()
303            .map(move |offset_range| self.get_field_by_offset(offset_range?));
304        Ok(iterator)
305    }
306}

308#[derive(Clone, Debug, PartialEq)]
309pub(crate) struct VariantObjectHeader {
310    field_offset_size: OffsetSizeBytes,
311    field_id_size: OffsetSizeBytes,
312    num_elements: usize,
313    field_ids_start_byte: usize,
314    field_offsets_start_byte: usize,
315    values_start_byte: usize,
316}
317
318impl VariantObjectHeader {
319    pub(crate) fn try_new(value: &[u8]) -> Result<Self, ArrowError> {
320        // Parse the header byte to get object parameters
321        let header = first_byte_from_slice(value)?;
322        let value_header = header >> 2;
323
324        let field_offset_size_minus_one = value_header & 0x03; // Last 2 bits
325        let field_id_size_minus_one = (value_header >> 2) & 0x03; // Next 2 bits
326        let is_large = value_header & 0x10; // 5th bit
327
328        let field_offset_size = OffsetSizeBytes::try_new(field_offset_size_minus_one)?;
329        let field_id_size = OffsetSizeBytes::try_new(field_id_size_minus_one)?;
330
331        // Determine num_elements size based on is_large flag
332        let num_elements_size = if is_large != 0 {
333            OffsetSizeBytes::Four
334        } else {
335            OffsetSizeBytes::One
336        };
337
338        // Parse num_elements
339        let num_elements = num_elements_size.unpack_usize(value, 1, 0)?;
340
341        // Calculate byte offsets for different sections
342        let field_ids_start_byte = 1 + num_elements_size as usize;
343        let field_offsets_start_byte = field_ids_start_byte + num_elements * field_id_size as usize;
344        let values_start_byte =
345            field_offsets_start_byte + (num_elements + 1) * field_offset_size as usize;
346
347        // Verify that the last field offset array entry is inside the value slice
348        let last_field_offset_byte =
349            field_offsets_start_byte + (num_elements + 1) * field_offset_size as usize;
350        if last_field_offset_byte > value.len() {
351            return Err(ArrowError::InvalidArgumentError(format!(
352                "Last field offset array entry at offset {} with length {} is outside the value slice of length {}",
353                last_field_offset_byte,
354                field_offset_size as usize,
355                value.len()
356            )));
357        }
358
359        // Verify that the value of the last field offset array entry fits inside the value slice
360        let last_field_offset =
361            field_offset_size.unpack_usize(value, field_offsets_start_byte, num_elements)?;
362        if values_start_byte + last_field_offset > value.len() {
363            return Err(ArrowError::InvalidArgumentError(format!(
364                "Last field offset value {} at offset {} is outside the value slice of length {}",
365                last_field_offset,
366                values_start_byte,
367                value.len()
368            )));
369        }
370        Ok(Self {
371            field_offset_size,
372            field_id_size,
373            num_elements,
374            field_ids_start_byte,
375            field_offsets_start_byte,
376            values_start_byte,
377        })
378    }
379
380    /// Returns the number of key-value pairs in this object
381    pub(crate) fn num_elements(&self) -> usize {
382        self.num_elements
383    }
384}

386#[derive(Clone, Debug, PartialEq)]
387pub struct VariantObject<'m, 'v> {
388    pub metadata: VariantMetadata<'m>,
389    pub value: &'v [u8],
390    header: VariantObjectHeader,
391}
392
393impl<'m, 'v> VariantObject<'m, 'v> {
394    pub fn try_new(metadata: VariantMetadata<'m>, value: &'v [u8]) -> Result<Self, ArrowError> {
395        Ok(Self {
396            metadata,
397            value,
398            header: VariantObjectHeader::try_new(value)?,
399        })
400    }
401
402    /// Returns the number of key-value pairs in this object
403    pub fn len(&self) -> usize {
404        self.header.num_elements()
405    }
406
407    /// Returns true if the object contains no key-value pairs
408    pub fn is_empty(&self) -> bool {
409        self.len() == 0
410    }
411
412    pub fn fields(&self) -> Result<impl Iterator<Item = (&'m str, Variant<'m, 'v>)>, ArrowError> {
413        let field_list = self.parse_field_list()?;
414        Ok(field_list.into_iter())
415    }
416
417    pub fn field(&self, name: &str) -> Result<Option<Variant<'m, 'v>>, ArrowError> {
418        // Binary search through the field IDs of this object to find the requested field name.
419        //
420        // NOTE: This does not require a sorted metadata dictionary, because the variant spec
421        // requires object field ids to be lexically sorted by their corresponding string values,
422        // and probing the dictionary for a field id is always O(1) work.
423        let (field_ids, field_offsets) = self.parse_field_arrays()?;
424        let search_result = try_binary_search_by(&field_ids, &name, |&field_id| {
425            self.metadata.get_field_by(field_id)
426        })?;
427
428        let Ok(index) = search_result else {
429            return Ok(None);
430        };
431        let start_offset = field_offsets[index];
432        let end_offset = field_offsets[index + 1];
433        let value_bytes = slice_from_slice(
434            self.value,
435            self.header.values_start_byte + start_offset
436                ..self.header.values_start_byte + end_offset,
437        )?;
438        let variant = Variant::try_new_with_metadata(self.metadata, value_bytes)?;
439        Ok(Some(variant))
440    }
441
442    /// Parse field IDs and field offsets arrays using the cached header
443    fn parse_field_arrays(&self) -> Result<(Vec<usize>, Vec<usize>), ArrowError> {
444        // Parse field IDs
445        let field_ids = (0..self.header.num_elements)
446            .map(|i| {
447                self.header.field_id_size.unpack_usize(
448                    self.value,
449                    self.header.field_ids_start_byte,
450                    i,
451                )
452            })
453            .collect::<Result<Vec<_>, _>>()?;
454        debug_assert_eq!(field_ids.len(), self.header.num_elements);
455
456        // Parse field offsets (num_elements + 1 entries)
457        let field_offsets = (0..=self.header.num_elements)
458            .map(|i| {
459                self.header.field_offset_size.unpack_usize(
460                    self.value,
461                    self.header.field_offsets_start_byte,
462                    i,
463                )
464            })
465            .collect::<Result<Vec<_>, _>>()?;
466        debug_assert_eq!(field_offsets.len(), self.header.num_elements + 1);
467
468        Ok((field_ids, field_offsets))
469    }
470
471    /// Parse all fields into a vector for iteration
472    fn parse_field_list(&self) -> Result<Vec<(&'m str, Variant<'m, 'v>)>, ArrowError> {
473        let (field_ids, field_offsets) = self.parse_field_arrays()?;
474
475        let mut fields = Vec::with_capacity(self.header.num_elements);
476
477        for i in 0..self.header.num_elements {
478            let field_id = field_ids[i];
479            let field_name = self.metadata.get_field_by(field_id)?;
480
481            let start_offset = field_offsets[i];
482            let value_bytes =
483                slice_from_slice(self.value, self.header.values_start_byte + start_offset..)?;
484            let variant = Variant::try_new_with_metadata(self.metadata, value_bytes)?;
485
486            fields.push((field_name, variant));
487        }
488
489        Ok(fields)
490    }
491}

493#[derive(Clone, Debug, PartialEq)]
494pub(crate) struct VariantListHeader {
495    offset_size: OffsetSizeBytes,
496    is_large: bool,
497    num_elements: usize,
498    first_offset_byte: usize,
499    first_value_byte: usize,
500}
501
502impl VariantListHeader {
503    pub(crate) fn try_new(value: &[u8]) -> Result<Self, ArrowError> {
504        // The 6 first bits to the left are the value_header and the 2 bits
505        // to the right are the basic type, so we shift to get only the value_header
506        let value_header = first_byte_from_slice(value)? >> 2;
507        let is_large = (value_header & 0x04) != 0; // 3rd bit from the right
508        let field_offset_size_minus_one = value_header & 0x03; // Last two bits
509        let offset_size = OffsetSizeBytes::try_new(field_offset_size_minus_one)?;
510
511        // The size of the num_elements entry in the array value_data is 4 bytes if
512        // is_large is true, otherwise 1 byte.
513        let num_elements_size = match is_large {
514            true => OffsetSizeBytes::Four,
515            false => OffsetSizeBytes::One,
516        };
517
518        // Skip the header byte to read the num_elements
519        let num_elements = num_elements_size.unpack_usize(value, 1, 0)?;
520        let first_offset_byte = 1 + num_elements_size as usize;
521
522        let overflow =
523            || ArrowError::InvalidArgumentError("Variant value_byte_length overflow".into());
524
525        // 1.  num_elements + 1
526        let n_offsets = num_elements.checked_add(1).ok_or_else(overflow)?;
527
528        // 2.  (num_elements + 1) * offset_size
529        let value_bytes = n_offsets
530            .checked_mul(offset_size as usize)
531            .ok_or_else(overflow)?;
532
533        // 3.  first_offset_byte + ...
534        let first_value_byte = first_offset_byte
535            .checked_add(value_bytes)
536            .ok_or_else(overflow)?;
537
538        // Verify that the last offset array entry is inside the value slice
539        let last_offset_byte = first_offset_byte + n_offsets * offset_size as usize;
540        if last_offset_byte > value.len() {
541            return Err(ArrowError::InvalidArgumentError(format!(
542                "Last offset array entry at offset {} with length {} is outside the value slice of length {}",
543                last_offset_byte,
544                offset_size as usize,
545                value.len()
546            )));
547        }
548
549        // Verify that the value of the last offset array entry fits inside the value slice
550        let last_offset = offset_size.unpack_usize(value, first_offset_byte, num_elements)?;
551        if first_value_byte + last_offset > value.len() {
552            return Err(ArrowError::InvalidArgumentError(format!(
553                "Last offset value {} at offset {} is outside the value slice of length {}",
554                last_offset,
555                first_value_byte,
556                value.len()
557            )));
558        }
559
560        Ok(Self {
561            offset_size,
562            is_large,
563            num_elements,
564            first_offset_byte,
565            first_value_byte,
566        })
567    }
568
569    /// Returns the number of elements in this list
570    pub(crate) fn num_elements(&self) -> usize {
571        self.num_elements
572    }
573
574    /// Returns the offset size in bytes
575    #[allow(unused)]
576    pub(crate) fn offset_size(&self) -> usize {
577        self.offset_size as _
578    }
579
580    /// Returns whether this is a large list
581    #[allow(unused)]
582    pub(crate) fn is_large(&self) -> bool {
583        self.is_large
584    }
585
586    /// Returns the byte offset where the offset array starts
587    pub(crate) fn first_offset_byte(&self) -> usize {
588        self.first_offset_byte
589    }
590
591    /// Returns the byte offset where the values start
592    pub(crate) fn first_value_byte(&self) -> usize {
593        self.first_value_byte
594    }

597/// Represents a variant array.
598///
599/// NOTE: The "list" naming differs from the variant spec -- which calls it "array" -- in order to be
600/// consistent with parquet and arrow type naming. Otherwise, the name would conflict with the
601/// `VariantArray : Array` we must eventually define for variant-typed arrow arrays.
602#[derive(Clone, Debug, PartialEq)]
603pub struct VariantList<'m, 'v> {
604    pub metadata: VariantMetadata<'m>,
605    pub value: &'v [u8],
606    header: VariantListHeader,
607}
608
609impl<'m, 'v> VariantList<'m, 'v> {
610    pub fn try_new(metadata: VariantMetadata<'m>, value: &'v [u8]) -> Result<Self, ArrowError> {
611        Ok(Self {
612            metadata,
613            value,
614            header: VariantListHeader::try_new(value)?,
615        })
616    }
617
618    /// Return the length of this array
619    pub fn len(&self) -> usize {
620        self.header.num_elements()
621    }
622
623    /// Is the array of zero length
624    pub fn is_empty(&self) -> bool {
625        self.len() == 0
626    }
627
628    pub fn values(&self) -> Result<impl Iterator<Item = Variant<'m, 'v>>, ArrowError> {
629        let len = self.len();
630        let values = (0..len)
631            .map(move |i| self.get(i))
632            .collect::<Result<Vec<_>, _>>()?;
633        Ok(values.into_iter())
634    }
635
636    pub fn get(&self, index: usize) -> Result<Variant<'m, 'v>, ArrowError> {
637        if index >= self.header.num_elements() {
638            return Err(ArrowError::InvalidArgumentError(format!(
639                "Index {} out of bounds for list of length {}",
640                index,
641                self.header.num_elements()
642            )));
643        }
644
645        // Skip header and num_elements bytes to read the offsets
646        let start_field_offset_from_first_value_byte = self.header.offset_size.unpack_usize(
647            self.value,
648            self.header.first_offset_byte(),
649            index,
650        )?;
651        let end_field_offset_from_first_value_byte = self.header.offset_size.unpack_usize(
652            self.value,
653            self.header.first_offset_byte(),
654            index + 1,
655        )?;
656
657        // Read the value bytes from the offsets
658        let variant_value_bytes = slice_from_slice(
659            self.value,
660            self.header.first_value_byte() + start_field_offset_from_first_value_byte
661                ..self.header.first_value_byte() + end_field_offset_from_first_value_byte,
662        )?;
663        let variant = Variant::try_new_with_metadata(self.metadata, variant_value_bytes)?;
664        Ok(variant)
665    }

699    /// Create a new `Variant` from metadata and value.
700    ///
701    /// # Example
702    /// ```
703    /// # use parquet_variant::{Variant, VariantMetadata};
704    /// let metadata = [0x01, 0x00, 0x00];

711    pub fn try_new(metadata: &'m [u8], value: &'v [u8]) -> Result<Self, ArrowError> {

716    /// Create a new variant with existing metadata
717    ///
718    /// # Example
719    /// ```
720    /// # use parquet_variant::{Variant, VariantMetadata};
721    /// let metadata = [0x01, 0x00, 0x00];
722    /// let value = [0x09, 0x48, 0x49];
723    /// // parse the header metadata first
724    /// let metadata = VariantMetadata::try_new(&metadata).unwrap();
725    /// assert_eq!(
726    ///   Variant::ShortString("HI"),
727    ///   Variant::try_new_with_metadata(metadata, &value).unwrap()
728    /// );
729    /// ```
730    pub fn try_new_with_metadata(
731        metadata: VariantMetadata<'m>,
732        value: &'v [u8],
733    ) -> Result<Self, ArrowError> {
734        let value_metadata = *first_byte_from_slice(value)?;
735        let value_data = slice_from_slice(value, 1..)?;
736        let new_self = match get_basic_type(value_metadata)? {
737            VariantBasicType::Primitive => match get_primitive_type(value_metadata)? {
738                VariantPrimitiveType::Null => Variant::Null,
739                VariantPrimitiveType::Int8 => Variant::Int8(decoder::decode_int8(value_data)?),
740                VariantPrimitiveType::Int16 => Variant::Int16(decoder::decode_int16(value_data)?),
741                VariantPrimitiveType::Int32 => Variant::Int32(decoder::decode_int32(value_data)?),
742                VariantPrimitiveType::Int64 => Variant::Int64(decoder::decode_int64(value_data)?),
743                VariantPrimitiveType::Decimal4 => {
744                    let (integer, scale) = decoder::decode_decimal4(value_data)?;
745                    Variant::Decimal4 { integer, scale }
746                }
747                VariantPrimitiveType::Decimal8 => {
748                    let (integer, scale) = decoder::decode_decimal8(value_data)?;
749                    Variant::Decimal8 { integer, scale }
750                }
751                VariantPrimitiveType::Decimal16 => {
752                    let (integer, scale) = decoder::decode_decimal16(value_data)?;
753                    Variant::Decimal16 { integer, scale }
754                }
755                VariantPrimitiveType::Float => Variant::Float(decoder::decode_float(value_data)?),
756                VariantPrimitiveType::Double => {
757                    Variant::Double(decoder::decode_double(value_data)?)
758                }
759                VariantPrimitiveType::BooleanTrue => Variant::BooleanTrue,
760                VariantPrimitiveType::BooleanFalse => Variant::BooleanFalse,
761                // TODO: Add types for the rest, once API is agreed upon
762                VariantPrimitiveType::Date => Variant::Date(decoder::decode_date(value_data)?),
763                VariantPrimitiveType::TimestampMicros => {
764                    Variant::TimestampMicros(decoder::decode_timestamp_micros(value_data)?)
765                }
766                VariantPrimitiveType::TimestampNtzMicros => {
767                    Variant::TimestampNtzMicros(decoder::decode_timestampntz_micros(value_data)?)
768                }
769                VariantPrimitiveType::Binary => {
770                    Variant::Binary(decoder::decode_binary(value_data)?)
771                }
772                VariantPrimitiveType::String => {
773                    Variant::String(decoder::decode_long_string(value_data)?)
774                }
775            },
776            VariantBasicType::ShortString => {
777                Variant::ShortString(decoder::decode_short_string(value_metadata, value_data)?)
778            }
779            VariantBasicType::Object => Variant::Object(VariantObject::try_new(metadata, value)?),
780            VariantBasicType::Array => Variant::List(VariantList::try_new(metadata, value)?),
781        };
782        Ok(new_self)
783    }
